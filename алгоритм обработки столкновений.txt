ответный сдвиг рекурсивная функция(предмет, сдвиг предмета)
{
	проверяем столкновения с движущимися предметами;
	если(столкновения с движущимися предметаеми есть)
	{
		рекурсивная функция(предмет, предположительный сдвиг предмета)
		повторяем вызов для всех соприкасающихся предметов
		суммируем ответный сдвиг;
		вернуть ответный сдвиг; 
	}
	иначе если(столкновения со стенами есть)
	{
		ответный сдвиг = -сдвиг предмета;
		вернуть ответный сдвиг;
	}
	иначе
	{
		сдвинуть себя;
		ответный сдвиг = 0;
		вернуть ответный сдвиг;
	}
}
